        \coordinate(coordTwo) at ({\xTwo},{\yTwo});

        \draw (#3,#4) -- (coordTwo);
        \begin{scope}[rotate around={60:(coordOne)}]
            \draw[red] (coordOne) -- (coordTwo);
        \end{scope}
